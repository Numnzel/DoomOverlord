class magicgravitymaelstrom : Actor {

	default {
		
		radius 6;
		height 12;
		speed 25;
		damage 110;
		ProjectileKickBack 0;
		RenderStyle "Add";
		Alpha 0.9;
		SeeSound "imp/attack";
		DeathSound "imp/shotx";
		PROJECTILE;
		+RANDOMIZE;
		+BLOODLESSIMPACT;
		+ROLLSPRITE;
		+HITMASTER;
	}

	states {
		spawn:
			FBAL ABCD 1 {
				A_SetRoll(random(0,359),SPF_INTERPOLATE);
			}
			Loop;
		death:
			BAL1 C 0;
			BAL1 C 0 A_GiveInventory("effectgravitymaelstrom", 1, AAPTR_MASTER);
			BAL1 CDE 6 A_SetScale(1.2);
			Stop;
	}
}

class effectgravitymaelstrom : CustomInventory {

	default {
		
		inventory.amount 1;
		+INVENTORY.AUTOACTIVATE;
	}
	
	states {
	
		pickup:
			TNT1 A 0;
			TNT1 A 0 A_Gravity();
			TNT1 A 0 A_SetGravity(30000);
			stop;
	}
}
